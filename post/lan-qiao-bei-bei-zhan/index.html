int bsearch_1(int l, int r)
{
    while (l < r)
    {
        int mid = l + r >> 1;
        if (check(mid)) r = mid;
        else l = mid + 1;
    }
    return l;
}

int bsearch_2(int l, int r)
{
    while (l < r)
    {
        //防死循环
        int mid = l + r + 1 >> 1;
        if (check(mid)) l = mid;
        else r = mid - 1;
    }
    return l;
}

#include<iostream>
using namespace std;
const int N=10;
int ans[N];
bool mark[N];
int n;
void dfs(int u)
{
    //"回头"的条件
    if(u==n)
    {
        for(int i=0;i<n;i++)cout<<ans[i]<<" ";
        puts("");
        return;
    }
    
    for(int i=1;i<=n;i++)
    {
        if(mark[i]==false)
        {
            mark[i]=true;
            #注意此处下标为u
            ans[u]=i;
            dfs(u+1);
            //复原,回溯
            mark[i]=false;
            ans[u]=0;
        }
    }
}

int main()
{
    cin>>n;
    dfs(0);
    return 0;
}

    if n == len(ans):

            state[i] = 0

动态规划技巧

动态规划:{
	状态表示:有限集合的最值,每次策略枚举的是一个子集(所有满足某种条件的集合)
	状态计算:一个化整为零的过程,划分为不同的子集(唯一,不遗漏),一般依据找最后一个不同点
	代码优化:做代码变形,尽量优化代码结构将高维降至低维
	{
	问题类型:
	选择类型问题(01背包);dp[i][j]  一般i表示选择前i个物品,j表示一个限制(容量,体积,时间等) f(i,j)一般分为俩个集合=>{
		不选择该物品的方案
		选择该物品的方案
		我们可以根据题目需求,对这俩个方案进行处理(求最大值,最小值等)
	}
	区间类型问题(石子合并):dp[i][j] 所有合并
	}
}

经典01背包问题(有一个容量为w的背包,有k件不同价值不同重量的物品,如何拿能使价值最高,选择问题)

#若把该函数记作dp(k,w):当背包容量为w,现有k件物品可以偷,所能偷到最大价值
#首先将物品按照价值进行降序排序,设第一件物品(排序后价值最高)重量为k1,价值为p1,
#状态转移方程如下
return max(dp(k-1,w-k1)+p1,dp(k-1,w))
#此时有俩种方案一种是选择当前物品(要求背包剩余容量大于等于当前物品重量),否则只能选择方案2,跳过此物品(不拿了)

具体实现(python),

n,m = map(int,input().split())
mylis = []

for i in range(n):
    #w代表重量,v代表价值
    w,v = map(int,input().split())
    mylis.append([w,v])

#表示当前还有k件物品,当前背包还有w容量,表示遍历到第i个物品
def dp(k,w,i):
    if i==len(mylis):
        return 0
    if k<=0 or w<=0:
        return 0
    #选择1是选择当前物品,选择2是跳过当前物品
    #如果选择当前物品,那么物品重量一定要小于当前背包重量
    if w>=mylis[i][0]:
        return max(dp(k-1,w-mylis[i][0],i+1)+mylis[i][1],dp(k-1,w,i+1))
    else:
        return dp(k-1,w,i+1)
print(dp(n,m,0))

#以上这种实现方式时间复杂度过高,可以考虑优化,拿一个dp数组存储每一步的结果
"""你可以使用一个二维数组dp来记录不同状态下的最大价值。

首先，定义dp[i][j]表示在前i个物品中选择若干个物品放入容量为j的背包中所能达到的最大价值。

然后，通过状态转移方程来更新dp数组：

当i=0时，dp[i][j]=0，表示没有物品可选，所以最大价值为0；
当j=0时，dp[i][j]=0，表示背包容量为0，所以最大价值为0；
dp[i][j] 表示剩余选择前i个物品是背包容量为j时,此时背包内物品的总最大价值
当j<mylis[i-1][0]时，dp[i][j]=dp[i-1][j]，表示背包容量不足以放入第i个物品，所以最大价值与前i-1个物品的最大价值相同；
当j>=mylis[i-1][0]时，dp[i][j]=max(dp[i-1][j], dp[i-1][j-mylis[i-1][0]]+mylis[i-1][1])，表示可以选择放入第i个物品或者不放入，取两者中的最大值。
"""

#优化如下,物品个数为n 容量为v
n,m = map(int,input().split())
mylis = []

for i in range(n):
    w,v = map(int,input().split())
    mylis.append([w,v])

dp = [[0]*(m+1) for _ in range(n+1)]

for i in range(1, n+1):
    for j in range(1, m+1):
        if j < mylis[i-1][0]:
            dp[i][j] = dp[i-1][j]
        else:
            dp[i][j] = max(dp[i-1][j], dp[i-1][j-mylis[i-1][0]]+mylis[i-1][1])

print(dp[n][m])

#区间dp和模板
dp = [[0 for _ in range(strlen)] for _ in range(strlen)]
#外层dp是每一轮字符串的区间长度,最小为2,最大为stelen
for len in range(2,strlen+1):
    #内层为区间判断
    left = 0
    while left+len-1<strlen:
        # 右边界left+len-1 
        right = left+len-1

        if int(s[left])>int(s[right]):
            dp[left][right] = 1
        if int(s[left])==int(s[right]):
            #向内寻找满足条件的子串

        left+=1

"""

        cnt+=1

可以看到当i>=2时且i满足(i-2)%4==0时不满足给定条件y^2-z^2条件,所以只需要计算在

可以用同区间-另一同区间的思想(前缀和的思想)

总个数
"""

count_jishu = math.ceil(R/2)-math.ceil((L-1)/2)

count_oushu = R//4-(L-1)//4
